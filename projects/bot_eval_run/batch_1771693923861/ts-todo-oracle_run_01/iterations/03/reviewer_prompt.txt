--- FILE: src/cli.ts ---
declare const require: any;
declare const process: any;
const fs = require('node:fs');
import { TaskStore, Task } from './store';

function help(): void {
  console.log(`
Task Manager CLI

Usage:
- add <title> --data <path>
- list --data <path>
- done <id> --data <path>
- remove <id> --data <path>
`);
}

function parseArgs(): { command: string, title?: string, id?: string, filePath: string } {
  const args = process.argv.slice(2);
  let command = args[0];
  let title: string | undefined;
  let id: string | undefined;
  let filePath: string | undefined;

  for (let i = 1; i < args.length; i++) {
    if (args[i] === '--data') {
      filePath = args[++i];
    } else if (!command) {
      command = args[i];
    } else if (!title && command === 'add') {
      title = args[i];
    } else if (!id && (command === 'done' || command === 'remove')) {
      id = args[i];
    }
  }

  return { command, title, id, filePath: filePath || 'tasks.json' };
}

function main(): void {
  const { command, title, id, filePath } = parseArgs();

  if (command === '--help') {
    help();
    return;
  }

  const taskStore = new TaskStore(filePath);
  taskStore.load();

  try {
    switch (command) {
      case 'add':
        if (!title) throw new Error('Title is required for add command');
        console.log(taskStore.add(title));
        break;
      case 'list':
        taskStore.list().forEach(task => console.log(`${task.id}: ${task.title} [${task.done ? 'Done' : 'Pending'}]`));
        break;
      case 'done':
        if (!id) throw new Error('ID is required for done command');
        taskStore.markDone(id);
        console.log(`Task ${id} marked as done.`);
        break;
      case 'remove':
        if (!id) throw new Error('ID is required for remove command');
        taskStore.remove(id);
        console.log(`Task ${id} removed.`);
        break;
      default:
        help();
    }
  } catch (error) {
    console.error(error.message);
  }
}

main();
