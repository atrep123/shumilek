import { argv } from 'process';
import TaskStore from './store';

const args = argv.slice(2);
const command = args[0];
const dataPath = args.find(arg => arg.startsWith('--data='))?.split('=')[1];

if (command === '--help' || !dataPath) {
  console.log('Usage: node dist/cli.js <command> --data=<path>');
  process.exit(0);
}

const taskStore = new TaskStore(dataPath);

switch (command) {
  case 'add': {
    const title = args.find(arg => !arg.startsWith('--'));
    if (!title) {
      console.log('Title is required for add command.');
      process.exit(1);
    }
    const task = taskStore.add(title);
    console.log(JSON.stringify({ ok: true, task }));
    break;
  }
  case 'list': {
    const tasks = taskStore.list();
    console.log(JSON.stringify({ ok: true, tasks }));
    break;
  }
  case 'done': {
    const id = args.find(arg => !arg.startsWith('--'));
    if (!id) {
      console.log('ID is required for done command.');
      process.exit(1);
    }
    const task = taskStore.done(id);
    console.log(JSON.stringify({ ok: true, task }));
    break;
  }
  case 'remove': {
    const id = args.find(arg => !arg.startsWith('--'));
    if (!id) {
      console.log('ID is required for remove command.');
      process.exit(1);
    }
    taskStore.remove(id);
    console.log(JSON.stringify({ ok: true }));
    break;
  }
  default:
    console.log(`Unknown command: ${command}`);
    process.exit(1);
}
