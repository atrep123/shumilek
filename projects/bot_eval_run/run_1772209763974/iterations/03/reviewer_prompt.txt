import unittest
from mini_ai.markov import MarkovChain
class TestMarkovChain(unittest.TestCase):
    def setUp(self):
        self.model = MarkovChain()
        self.text = 'hello world'
        self.model.train(self.text)

    def test_train(self):
        self.assertIn('h', self.model.chain)
        self.assertIn('e', self.model.chain['h'])
        self.assertIn('l', self.model.chain['e'])
        self.assertIn('o', self.model.chain['l'])
        self.assertIn(' ', self.model.chain['o'])
        self.assertIn('w', self.model.chain[' '])

    def test_generate(self):
        generated_text = self.model.generate(10)
        self.assertEqual(len(generated_text), 10)
        for char in generated_text:
            self.assertTrue(char in self.text or char == '')

if __name__ == '__main__':
    unittest.main()
