from mini_ai.markov import MarkovChain

def _save_model(model: MarkovChain, path: str) -> None:
    with open(path, 'w', encoding='utf-8') as f:
        json.dump(model.to_dict(), f, ensure_ascii=False)

def _load_model(path: str) -> MarkovChain:
    with open(path, 'r', encoding='utf-8') as f:
        data = json.load(f)
    return MarkovChain.from_dict(data)

import random
from collections import defaultdict, Counter

class MarkovChain:
    def __init__(self, order: int = 1):
        self.order = order
        self.chain = defaultdict(Counter)

    def train(self, text: str) -> None:
        for i in range(len(text) - self.order):
            prefix = text[i:i + self.order]
            suffix = text[i + self.order]
            self.chain[prefix][suffix] += 1

    def generate(self, length: int, seed: str | None = None, random_seed: int | None = None) -> str:
        if random_seed is not None:
            random.seed(random_seed)

        if seed is None:
            prefix = random.choice(list(self.chain.keys()))
        else:
            prefix = seed[-self.order:] if len(seed) >= self.order else seed.ljust(self.order, ' ')

        result = list(prefix)
        for _ in range(length - self.order):
            suffixes = self.chain.get(prefix)
            if not suffixes:
                break
            next_char = random.choices(list(suffixes.keys()), weights=suffixes.values())[0]
            result.append(next_char)
            prefix = ''.join(result[-self.order:])

        return ''.join(result)

    def to_dict(self) -> dict:
        return {
            'order': self.order,
            'chain': {k: dict(v) for k, v in self.chain.items()}
        }

    @classmethod
    def from_dict(cls, data: dict) -> MarkovChain:
        instance = cls(order=data['order'])
        instance.chain = defaultdict(Counter, {k: Counter(v) for k, v in data['chain'].items()})
        return instance
