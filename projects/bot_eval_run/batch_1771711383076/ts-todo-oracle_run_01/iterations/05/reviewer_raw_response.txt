    "content": "// src/store.ts\n\ninterface Task {\n  id: string;\n  title: string;\n  done: boolean;\n  createdAt: string;\n}\n\nconst tasks: Task[] = [];\n\nclass TaskStore {\n  constructor(private dataPath: string) {}\n\n  list(): Task[] {\n    return [...tasks].sort((a, b) => new Date(b.createdAt) - new Date(a.createdAt));\n  }\n\n  add(title: string): Task {\n    const id = Math.random().toString(36).substring(2);\n    const createdAt = new Date().toISOString();\n    tasks.push({ id, title, done: false, createdAt });\n    fs.writeFileSync(this.dataPath, JSON.stringify(tasks), { encoding: 'utf-8' });\n    return this.list()[0];\n  }\n\n  done(id: string): Task {\n    const task = tasks.find(t => t.id === id);\n    if (!task) throw new Error('Task not found');\n    task.done = true;\n    fs.writeFileSync(this.dataPath, JSON.stringify(tasks), { encoding: 'utf-8' });\n    return task;\n  }\n\n  remove(id: string): Task {\n    const index = tasks.findIndex(t => t.id === id);\n    if (index < 0) throw new Error('Task not found');\n    const removedTask = tasks.splice(index, 1)[0];\n    fs.writeFileSync(this.dataPath, JSON.stringify(tasks), { encoding: 'utf-8' });\n    return removedTask;\n  }\n}\n\nexport default TaskStore;"
