Vytvor realisticky TypeScript projekt (bez externich zavislosti), ktery zpracuje CSV soubory a projde oracle testy.

POZADAVKY / KONTRAKT:
- Zadne externi npm balicky. Pouze Node.js builtin moduly.
- VYSLOVNE ZAKAZANO: csv-parse, papaparse, fast-csv, commander, yargs (a jine externi balicky).
- README.md je povinny.
- Nevytvarej `dist/` ve vystupu (generuje ho `tsc`).
- `package.json` nesmi mit `"type": "module"` (pouzij `commonjs`).
- V TS nepouzivej externi helper knihovny. Pro kompatibilitu muzes pouzit `declare const require: any; declare const process: any;`.
- Pouzij `const fs = require("node:fs")` (ne `declare const fs = ...`).
- Musi existovat `tsconfig.json` a kompilace do `dist/` (CommonJS) pres `tsc -p tsconfig.json`.

- `src/csv.ts` musi exportovat `CsvParser` a `CsvFilter`:

  CsvParser:
  - constructor(options?: { delimiter?: string }) — vychozi oddelovac je carka
  - parse(text: string): Record<string, string>[] — parsuje CSV text s prvnim radkem jako hlavicka; vraci pole objektu
  - stringify(rows: Record<string, string>[]): string — serializuje pole objektu zpet do CSV (s hlavickou)
  - Musi podporovat quoted fields (uvozovky kolem poli obsahujicich oddelovac nebo uvozovky, escaped "" uvnitr)

  CsvFilter:
  - constructor(rows: Record<string, string>[])
  - where(predicate: (row: Record<string, string>) => boolean): Record<string, string>[] — filtruje radky; MUSI vratit pole, nikdy CsvFilter
  - select(columns: string[]): Record<string, string>[] — vybere jen urcite sloupce; MUSI vratit pole, nikdy CsvFilter
  - sortBy(column: string): Record<string, string>[] — seradi radky podle hodnot sloupce (ascending, string compare); MUSI vratit pole, nikdy CsvFilter
  - count(): number — pocet radku
  - ZAKAZANO: fluent/chaining API pro CsvFilter (`where(...): CsvFilter`, `select(...): CsvFilter`, `sortBy(...): CsvFilter`).
  - `new CsvFilter(rows)` slouzi jen jako obal pro `count()`. Metody `where`, `select`, `sortBy` vraceji finalni pole radku, ne dalsi instanci.

- `src/cli.ts` musi po kompilaci vytvorit spustitelny CLI v `dist/cli.js`:
  - `--help` (exit code 0)
  - `parse --input <soubor>` — nacte CSV soubor, vypise JSON pole radku na stdout
  - `stats --input <soubor>` — nacte CSV soubor, vypise statistiky (pocet radku, sloupce) na stdout

Technicka poznamka: abys nemusel resit @types/node, v TS nepouzivej `import fs from ...`.
Pouzij `declare const require: any; declare const process: any;` a nacti builtin pres `const fs = require("node:fs")` atd.

const test = require('node:test');
const assert = require('node:assert/strict');
const fs = require('node:fs');
const path = require('node:path');
const os = require('node:os');
const { spawnSync } = require('node:child_process');

function makeTempDir() {
  return fs.mkdtempSync(path.join(os.tmpdir(), 'bot-eval-ts-csv-'));
}

function runCli(workspaceDir, args) {
  const cliPath = path.join(workspaceDir, 'dist', 'cli.js');
  const res = spawnSync(process.execPath, [cliPath, ...args], { cwd: workspaceDir, encoding: 'utf8' });
  return { code: res.status ?? 1, stdout: res.stdout || '', stderr: res.stderr || '' };
}

test('CsvParser: parse simple CSV', () => {
  const { CsvParser } = require('../dist/csv.js');
  assert.equal(typeof CsvParser, 'function');

  const parser = new CsvParser();
  const rows = parser.parse('name,age\nAlice,30\nBob,25\n');

  assert.equal(rows.length, 2);
  assert.equal(rows[0].name, 'Alice');
  assert.equal(rows[0].age, '30');
  assert.equal(rows[1].name, 'Bob');
  assert.equal(rows[1].age, '25');
});

test('CsvParser: parse CSV with quoted fields', () => {

  const parser = new CsvParser();

  const csv = 'name,bio\n"Alice","Likes ""coding"" and cats"\n"Bob","Hello, world"\n';

test('CsvParser: stringify rows to CSV', () => {
  const { CsvParser } = require('../dist/csv.js');
  const parser = new CsvParser();

  const rows = [
    { name: 'Alice', age: '30' },
    { name: 'Bob', age: '25' }
  ];

  const csv = parser.stringify(rows);
  assert.ok(csv.includes('name'));
  assert.ok(csv.includes('age'));
  assert.ok(csv.includes('Alice'));
  assert.ok(csv.includes('30'));
  assert.ok(csv.includes('Bob'));

  // Round-trip: parse the generated CSV back
  const parsed = parser.parse(csv);
  assert.equal(parsed.length, 2);
  assert.equal(parsed[0].name, 'Alice');
  assert.equal(parsed[1].age, '25');
});

test('CsvParser: stringify quotes fields with commas', () => {
  const { CsvParser } = require('../dist/csv.js');
  const parser = new CsvParser();

  const rows = [{ name: 'Alice, Bob', age: '30' }];
  const csv = parser.stringify(rows);
  assert.ok(csv.includes('"Alice, Bob"'));
});

test('CsvFilter: filter rows by predicate', () => {
  const { CsvParser, CsvFilter } = require('../dist/csv.js');
  assert.equal(typeof CsvFilter, 'function');

  const parser = new CsvParser();
  const rows = parser.parse('name,age\nAlice,30\nBob,25\nCharlie,35\n');

  const filter = new CsvFilter(rows);
  const adults = filter.where((row) => parseInt(row.age, 10) >= 30);

  assert.equal(adults.length, 2);
  assert.equal(adults[0].name, 'Alice');
  assert.equal(adults[1].name, 'Charlie');
});

test('CsvFilter: select specific columns', () => {
  const { CsvParser, CsvFilter } = require('../dist/csv.js');

  const parser = new CsvParser();
  const rows = parser.parse('name,age,city\nAlice,30,Prague\nBob,25,Brno\n');

  const filter = new CsvFilter(rows);
  const projected = filter.select(['name', 'city']);

  assert.equal(projected.length, 2);
  assert.deepEqual(Object.keys(projected[0]).sort(), ['city', 'name']);
  assert.equal(projected[0].name, 'Alice');
  assert.equal(projected[0].city, 'Prague');
});

test('CsvFilter: sort rows', () => {
  const { CsvParser, CsvFilter } = require('../dist/csv.js');

  const parser = new CsvParser();
  const rows = parser.parse('name,age\nCharlie,35\nAlice,30\nBob,25\n');

  const filter = new CsvFilter(rows);
  const sorted = filter.sortBy('name');

  assert.equal(sorted[0].name, 'Alice');
  assert.equal(sorted[1].name, 'Bob');
  assert.equal(sorted[2].name, 'Charlie');
});

test('CsvFilter: count rows', () => {
  const { CsvParser, CsvFilter } = require('../dist/csv.js');

  const parser = new CsvParser();
  const rows = parser.parse('name,age\nAlice,30\nBob,25\n');

  const filter = new CsvFilter(rows);
  assert.equal(filter.count(), 2);
});

test('CLI: --help exits with code 0', (t) => {
  const cliPath = path.join(__dirname, '..', 'dist', 'cli.js');
  if (!fs.existsSync(cliPath)) {
    t.skip('dist/cli.js not found');

  }

  assert.equal(result.code, 0);
});

test('CLI: parse command reads file and outputs JSON', (t) => {
  const cliPath = path.join(__dirname, '..', 'dist', 'cli.js');
  if (!fs.existsSync(cliPath)) {
    t.skip('dist/cli.js not found');
    return;
  }

  const dir = makeTempDir();
  const csvFile = path.join(dir, 'data.csv');
  fs.writeFileSync(csvFile, 'name,age\nAlice,30\nBob,25\n');

  const result = runCli(path.join(__dirname, '..'), ['parse', '--input', csvFile]);
  assert.equal(result.code, 0);

  const stdout = result.stdout.trim();
  const start = stdout.indexOf('[');
  const end = stdout.lastIndexOf(']');
  assert.ok(start >= 0 && end > start, 'Expected JSON array on stdout');
  const parsed = JSON.parse(stdout.slice(start, end + 1));
  assert.equal(parsed.length, 2);
  assert.equal(parsed[0].name, 'Alice');
});
