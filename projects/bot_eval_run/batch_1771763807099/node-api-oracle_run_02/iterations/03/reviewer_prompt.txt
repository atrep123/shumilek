// src/server.js
const http = require('http');
const fs = require('fs');
const path = require('path');
const uuid = require('uuid').v4;

function loadTodos(dataPath) {
  if (fs.existsSync(dataPath)) {
    return JSON.parse(fs.readFileSync(dataPath, 'utf8')) || [];
  }
  return [];
}

function saveTodos(todos, dataPath) {
  fs.writeFileSync(dataPath, JSON.stringify(todos, null, 2), 'utf8');
}

function createServer(dataPath) {
  const todos = loadTodos(dataPath);

  return http.createServer((req, res) => {
    let body = '';
    req.on('data', chunk => {
      body += chunk.toString();
    });

    req.on('end', () => {
      if (req.method === 'GET' && req.url === '/health') {
        res.writeHead(200, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify({ status: 'healthy' }));
      } else if (req.method === 'GET' && req.url === '/todos') {
        res.writeHead(200, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify({ ok: true, todos }));
      } else if (req.method === 'POST' && req.url === '/todos') {
        const todo = JSON.parse(body);
        if (!todo.title) {
          res.writeHead(400, { 'Content-Type': 'application/json' });
          res.end(JSON.stringify({ ok: false, error: 'Title is required' }));
          return;
        }
        const newTodo = {
          id: uuid(),
          title: todo.title,
          done: false,
          createdAt: new Date().toISOString()
        };
        todos.push(newTodo);
        saveTodos(todos, dataPath);
        res.writeHead(201, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify({ ok: true, todo: newTodo }));
      } else if (req.method === 'GET') {
        const id = req.url.split('/').pop();
        const todo = todos.find(t => t.id === id);
        if (!todo) {
          res.writeHead(404, { 'Content-Type': 'application/json' });
          res.end(JSON.stringify({ ok: false, error: 'Todo not found' }));
          return;
        }
        res.writeHead(200, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify({ ok: true, todo }));
      } else if (req.method === 'PATCH') {
        const id = req.url.split('/').pop();
        let todo = todos.find(t => t.id === id);
        if (!todo) {
          res.writeHead(404, { 'Content-Type': 'application/json' });
          res.end(JSON.stringify({ ok: false, error: 'Todo not found' }));
          return;
        }
        const update = JSON.parse(body);
        if (update.done !== undefined) {
          todo.done = update.done;
          if (todo.done) {
            todo.doneAt = new Date().toISOString();
          }
        }
        saveTodos(todos, dataPath);
        res.writeHead(200, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify({ ok: true, todo }));
      } else if (req.method === 'DELETE') {
        const id = req.url.split('/').pop();
        const index = todos.findIndex(t => t.id === id);
        if (index === -1) {
          res.writeHead(404, { 'Content-Type': 'application/json' });
          res.end(JSON.stringify({ ok: false, error: 'Todo not found' }));
          return;
        }
        todos.splice(index, 1);
        saveTodos(todos, dataPath);
        res.writeHead(200, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify({ ok: true }));
      } else {
        res.writeHead(404, { 'Content-Type': 'text/plain' });
        res.end('Not Found');
      }
    });
  });
}
